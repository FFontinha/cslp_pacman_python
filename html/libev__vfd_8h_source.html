81 /*
82  * Given an OS handle finds or allocates a virtual fd
83  * Returns -1 on failure and sets Python exception if pyexc is non-zero
84  */
85 static int vfd_open_(vfd_socket_t handle, int pyexc)
86 {
87  VFD_GIL_DECLARE;
88  int fd = -1;
89  unsigned long arg;
90  PyObject* key = NULL;
91  PyObject* value;
92 
93  if (!pyexc) {
94  VFD_GIL_ENSURE;
95  }
96  if (ioctlsocket(handle, FIONREAD, &arg) != 0) {
97  if (pyexc)
98  PyErr_Format(PyExc_IOError,
99 #ifdef _WIN64
100  "%lld is not a socket (files are not supported)",
101 #else
102  "%ld is not a socket (files are not supported)",
103 #endif
104  handle);
105  goto done;
106  }
107  if (vfd_map == NULL) {
108  vfd_map = PyDict_New();
109  if (vfd_map == NULL)
110  goto done;
111  }
112  key = vfd_socket_object(handle);
113  /* check if it's already in the dict */
114  value = PyDict_GetItem(vfd_map, key);
115  if (value != NULL) {
116  /* is it safe to use PyInt_AS_LONG(value) here? */
117  fd = PyInt_AsLong(value);
118  if (fd >= 0) {
119  ++vfd_entries[fd].count;
120  goto done;
121  }
122  }
123  /* use the free entry, if available */
124  if (vfd_next >= 0) {
125  fd = vfd_next;
126  vfd_next = vfd_entries[fd].next;
127  VFD_LOCK_ENTER;
128  goto allocated;
129  }
130  /* check if it would be out of bounds */
131  if (vfd_num >= FD_SETSIZE) {
132  /* libev's select doesn't support more that FD_SETSIZE fds */
133  if (pyexc)
134  PyErr_Format(PyExc_IOError, "cannot watch more than %d sockets", (int)FD_SETSIZE);
135  goto done;
136  }
137  /* allocate more space if needed */
138  VFD_LOCK_ENTER;
139  if (vfd_num >= vfd_max) {
140  int newsize = vfd_max + VFD_INCREMENT;
141  vfd_entry* entries = PyMem_Realloc(vfd_entries, sizeof(vfd_entry) * newsize);
142  if (entries == NULL) {
143  VFD_LOCK_LEAVE;
144  if (pyexc)
145  PyErr_NoMemory();
146  goto done;
147  }
148  vfd_entries = entries;
149  vfd_max = newsize;
150  }
151  fd = vfd_num++;
152 allocated:
153  /* vfd_lock must be acquired when entering here */
154  vfd_entries[fd].handle = handle;
155  vfd_entries[fd].count = 1;
156  VFD_LOCK_LEAVE;
157  value = PyInt_FromLong(fd);
158  PyDict_SetItem(vfd_map, key, value);
159  Py_DECREF(value);
160 done:
161  Py_XDECREF(key);
162  if (!pyexc) {
163  VFD_GIL_RELEASE;
164  }
165  return fd;
166 }

171 static void vfd_free_(int fd, int needclose)
172 {
173  VFD_GIL_DECLARE;
174  PyObject* key;
175 
176  if (needclose) {
177  VFD_GIL_ENSURE;
178  }
179  if (fd < 0 || fd >= vfd_num)
180  goto done; /* out of bounds */
181  if (vfd_entries[fd].count <= 0)
182  goto done; /* free entry, ignore */
183  if (!--vfd_entries[fd].count) {
184  /* fd has just been freed */
185  vfd_socket_t handle = vfd_entries[fd].handle;
186  vfd_entries[fd].handle = -1;
187  vfd_entries[fd].next = vfd_next;
188  vfd_next = fd;
189  if (needclose)
190  closesocket(handle);
191  /* vfd_map is assumed to be != NULL */
192  key = vfd_socket_object(handle);
193  PyDict_DelItem(vfd_map, key);
194  Py_DECREF(key);
195  }
196 done:
197  if (needclose) {
198  VFD_GIL_RELEASE;
199  }
200 }
